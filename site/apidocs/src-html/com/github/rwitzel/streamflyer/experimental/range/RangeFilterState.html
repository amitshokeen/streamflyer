016package com.github.rwitzel.streamflyer.experimental.range;
017
018import java.util.regex.MatchResult;
019
020import com.github.rwitzel.streamflyer.core.AfterModification;
021import com.github.rwitzel.streamflyer.experimental.stateful.State;
022import com.github.rwitzel.streamflyer.experimental.stateful.util.RegexTransitionState;
023import com.github.rwitzel.streamflyer.regex.RegexModifier;
024
025/**
026 * An internal class used by {@link RangeFilterModifier}.
027 * <p>
028 * This state has exactly one other succeeding state. That state will be the next state if the token this state refers
029 * to is found.
030 * <p>
031 * Additionally, you can define
032 * <ul>
033 * <li>whether the characters that appear before the found token shall be deleted from the stream,
034 * <li>whether the token itself shall be deleted from the stream.
035 * 
036 * @author rwoo
037 * @since 14.09.2011
038 */
039public class RangeFilterState extends RegexTransitionState {
040
041    /**
042     * The next state if the token the state refers to is found.
043     */
044    private State nextState;
045
046    /**
047     * True if the characters before the token the state refers to shall be deleted from the character buffer.
048     */
049    private boolean deleteCharactersBeforeToken;
050
051    /**
052     * True if the the token the state refers to shall be deleted from the character buffer.
053     */
054    private boolean deleteToken;
055
056    public RangeFilterState(RegexModifier nextTokensRegexModifier, boolean deleteCharactersBeforeToken,
057            boolean deleteToken) {
058        super(nextTokensRegexModifier);
059
060        this.deleteCharactersBeforeToken = deleteCharactersBeforeToken;
061        this.deleteToken = deleteToken;
062    }
063
064    /**
065     * @see com.github.rwitzel.streamflyer.experimental.stateful.util.RegexTransitionState#findState(java.lang.StringBuilder,
066     *      int, java.util.regex.MatchResult)
067     */
068    @Override
069    protected State findState(StringBuilder characterBuffer, int firstModifiableCharacterInBuffer,
070            MatchResult matchResult) {
071        return nextState;
072    }
073
074    /**
075     * @see com.github.rwitzel.streamflyer.experimental.stateful.util.RegexTransitionState#modifyBuffer(java.lang.StringBuilder,
076     *      int, java.util.regex.MatchResult)
077     */
078    @Override
079    protected int modifyBuffer(StringBuilder characterBuffer, int firstModifiableCharacterInBuffer,
080            MatchResult matchResult) {
081
082        int shiftMatchResult = 0;
083
084        // delete characters before token
085        if (deleteCharactersBeforeToken) {
086            characterBuffer.delete(firstModifiableCharacterInBuffer, matchResult.start());
087            shiftMatchResult += matchResult.start() - firstModifiableCharacterInBuffer;
088        }
089
090        // delete token
091        if (deleteToken) {
092            characterBuffer.delete(matchResult.start() - shiftMatchResult, matchResult.end() - shiftMatchResult);
093            shiftMatchResult += matchResult.end() - matchResult.start();
094        }
095
096        return matchResult.end() - shiftMatchResult;
097    }
098
099    /**
100     * @see com.github.rwitzel.streamflyer.experimental.stateful.util.RegexTransitionState#processWithoutMatch(java.lang.StringBuilder,
101     *      int, boolean, com.github.rwitzel.streamflyer.core.AfterModification)
102     */
103    @Override
104    protected AfterModification processWithoutMatch(StringBuilder characterBuffer,
105            int firstModifiableCharacterInBuffer, boolean endOfStreamHit, AfterModification afterModification) {
106
107        if (deleteCharactersBeforeToken && afterModification.getNumberOfCharactersToSkip() > 0) {
108
109            // delete the characters to skip
110            characterBuffer.delete(firstModifiableCharacterInBuffer, firstModifiableCharacterInBuffer
111                    + afterModification.getNumberOfCharactersToSkip());
112
113            // modify again immediately (??? TODO please review:
114            // MinimumLengthOfLookBehind should be recalculated? and so on...
115            return new AfterModification(0, true, firstModifiableCharacterInBuffer,
116                    afterModification.getNewNumberOfChars());
117        } else {
118            return afterModification;
119        }
120    }
121
122    //
123    // setter
124    //
125
126    /**
127     * @param state
128     *            The {@link #nextState} to set.
129     */
130    public void setNextState(State state) {
131        this.nextState = state;
132    }
133
134    //
135    // override Object.*
136    //
137
138    /**
139     * @see java.lang.Object#toString()
140     */
141    @Override
142    public String toString() {
143        StringBuilder builder = new StringBuilder();
144        builder.append("RangeFilterState [nextState=");
145        // builder.append(nextState);
146        builder.append("<nextState>"); // no endless loop, please
147        builder.append(", \ndeleteCharactersBeforeToken=");
148        builder.append(deleteCharactersBeforeToken);
149        builder.append(", \ndeleteToken=");
150        builder.append(deleteToken);
151        builder.append(", \ntoString()=");
152        builder.append(super.toString());
153        builder.append("]");
154        return builder.toString();
155    }
156}
