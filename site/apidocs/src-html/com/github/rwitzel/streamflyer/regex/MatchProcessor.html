016package com.github.rwitzel.streamflyer.regex;
017
018import java.util.regex.MatchResult;
019
020import com.github.rwitzel.streamflyer.core.Modifier;
021
022/**
023 * This interface is used by {@link RegexModifier}. This interface defines what to do if the regular expression matches
024 * some text in the stream.
025 * 
026 * @author rwoo
027 * @since 18.06.2011
028 */
029public interface MatchProcessor {
030
031    /**
032     * Processes the given match. The match processor is allowed to modify the given character buffer but it must not
033     * modify the characters before the first modifiable character in the buffer because the characters before that
034     * position are considered unmodifiable.
035     * 
036     * @param characterBuffer
037     *            Must not be <code>null</code>.
038     * @param firstModifiableCharacterInBuffer
039     *            See the parameter with the same name in {@link Modifier#modify(StringBuilder, int, boolean)}.
040     * @param matchResult
041     *            The match that is found in the given character buffer. Must not be <code>null</code>.
042     * @return Returns an {@link MatchProcessorResult object} that describes where to continue the matching.
043     */
044    public MatchProcessorResult process(StringBuilder characterBuffer, int firstModifiableCharacterInBuffer,
045            MatchResult matchResult);
046
047}
