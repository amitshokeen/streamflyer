017package com.github.rwitzel.streamflyer.util;
018
019import java.io.BufferedInputStream;
020import java.io.BufferedReader;
021import java.io.IOException;
022import java.io.InputStream;
023import java.io.Reader;
024
025import org.apache.commons.io.input.XmlStreamReader;
026
027import com.github.rwitzel.streamflyer.core.Modifier;
028import com.github.rwitzel.streamflyer.core.ModifyingReader;
029import com.github.rwitzel.streamflyer.regex.RegexModifier;
030import com.github.rwitzel.streamflyer.xml.InvalidXmlCharacterModifier;
031import com.github.rwitzel.streamflyer.xml.XmlVersionModifier;
032import com.github.rwitzel.streamflyer.xml.XmlVersionReader;
033
034/**
035 * Provides short-cuts to create {@link ModifyingReader modifying readers} using defaults.
036 * 
037 * @author rwoo
038 * @since 27.06.2011
039 */
040public class ModifyingReaderFactory {
041
042    public ModifyingReader createInvalidXmlCharacterRemovingReader(InputStream xmlStream) throws IOException {
043        return createInvalidXmlCharacterReplacingReader(xmlStream, "");
044    }
045
046    public ModifyingReader createInvalidXmlCharacterReplacingReader(InputStream xmlStream, String replacement)
047            throws IOException {
048
049        // buffer stream
050        // (is this really necessary to get optimal performance?)
051        if (!(xmlStream instanceof BufferedInputStream)) {
052            xmlStream = new BufferedInputStream(xmlStream);
053        }
054
055        // get the XML version
056        XmlStreamReader xmlReader = new XmlStreamReader(xmlStream);
057        XmlVersionReader xmlVersionReader = new XmlVersionReader(xmlReader);
058        String xmlVersion = xmlVersionReader.getXmlVersion();
059
060        // what kind of replacement?
061        boolean dollarZero;
062        if (replacement.contains("$0")) {
063            dollarZero = true;
064        } else {
065            dollarZero = false;
066        }
067
068        // create the reader that replaces invalid XML characters
069        Modifier modifier = new InvalidXmlCharacterModifier(8192, replacement, xmlVersion, dollarZero);
070        return new ModifyingReader(xmlVersionReader, modifier);
071    }
072
073    public ModifyingReader createXmlVersionModifyingReader(InputStream xmlStream, String newXmlVersion)
074            throws IOException {
075
076        // buffer stream
077        // (is this really necessary to get optimal performance?)
078        if (!(xmlStream instanceof BufferedInputStream)) {
079            xmlStream = new BufferedInputStream(xmlStream);
080        }
081
082        XmlStreamReader xmlReader = new XmlStreamReader(xmlStream);
083        XmlVersionReader xmlVersionReader = new XmlVersionReader(xmlReader);
084
085        // create the reader that replaces the XML version in prolog
086        Modifier modifier = new XmlVersionModifier(newXmlVersion, 8192);
087        return new ModifyingReader(xmlVersionReader, modifier);
088    }
089
090    public ModifyingReader createRegexModifyingReader(Reader input, String regex, String replacement) {
091        return createRegexModifyingReader(input, regex, 0, replacement, 0, 8192);
092    }
093
094    public ModifyingReader createRegexModifyingReader(Reader input, String regex, int flags, String replacement,
095            int minimumLengthOfLookBehind, int requestedCapacityOfCharacterBuffer) {
096
097        // buffer stream
098        // (is this really necessary to get optimal performance?)
099        if (!(input instanceof BufferedReader)) {
100            input = new BufferedReader(input);
101        }
102
103        // create modifier
104        Modifier modifier = new RegexModifier(regex, flags, replacement, minimumLengthOfLookBehind,
105                requestedCapacityOfCharacterBuffer);
106
107        // create and return reader
108        return new ModifyingReader(input, modifier);
109    }
110}
