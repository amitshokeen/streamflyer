017package com.github.rwitzel.streamflyer.core;
018
019import java.io.BufferedReader;
020import java.io.IOException;
021import java.io.Reader;
022import java.util.ArrayList;
023import java.util.Collections;
024import java.util.HashMap;
025import java.util.List;
026import java.util.Map;

028/**
029 * A {@link Reader} that allows a {@link Modifier} to modify the characters in the underlying stream.
030 * <p>
031 * This class is not synchronized.
032 * <p>
033 * {@link #markSupported() Marking} is not supported by this reader.
034 * 
035 * @author rwoo
036 * @since 06.05.2011
037 */
038public class ModifyingReader extends Reader {
039
040    //
041    // injected properties
042    //
043
044    /**
045     * The reader that the reads from the underlying character stream.
046     */
047    protected Reader delegate;
048
049    /**
050     * The modifier thats replaces, deletes, inserts characters of the underlying stream, and manages the buffer size.
051     */
052    private Modifier modifier;
053
054    //
055    // properties that represent the mutable state
056    //
057
058    /**
059     * At its begin the buffer contains unmodifiable characters (might be needed for look-behind matches as known from
060     * regular expression matching). After the unmodifiable characters the buffer holds the modifiable characters. See
061     * {@link #firstModifiableCharacterInBuffer}.
062     */
063    private StringBuilder characterBuffer;
064
065    /**
066     * The position of the first character in the {@link #characterBuffer input buffer} that is modifiable.
067     * <p>
068     * This character and the following characters in {@link #characterBuffer} can be modified by the {@link #modifier}.
069     */
070    private int firstModifiableCharacterInBuffer;
071
072    /**
073     * The value is taken from {@link AfterModification#getNewMinimumLengthOfLookBehind()} after
074     * {@link Modifier#modify(StringBuilder, int, boolean)} is called. The value is <em>requested</em> by the
075     * {@link Modifier}.
076     * <p>
077     * If this value is greater than {@link #firstModifiableCharacterInBuffer}, the modifier is probably faulty.
078     */
079    private int minimumLengthOfLookBehind;
080
081    /**
082     * The value is the sum of {@link AfterModification#getNewMinimumLengthOfLookBehind()} and
083     * {@link AfterModification#getNewNumberOfChars()} after {@link Modifier#modify(StringBuilder, int, boolean)} is
084     * called. The value is <em>requested</em> by the {@link Modifier}.
085     */
086    private int requestedNumCharactersInBuffer;
087
088    /**
089     * The value is initially taken from {@link AfterModification#getNumberOfCharactersToSkip()} after
090     * {@link Modifier#modify(StringBuilder, int, boolean)} is called. Then during reading characters from the reader
091     * this value is decremented.
092     */
093    private int numberOfCharactersToSkip;
094
095    /**
096     * True if the end of stream was detected.
097     */
098    private boolean endOfStreamHit = false;
099
100    /**
101     * The holds the last {@link AfterModification} provided by the {@link #modifier}. This property serves debugging
102     * purposes only.
103     */
104    private AfterModification lastAfterModificationForDebuggingOnly = null;
105
106    /**
107     * @param reader
108     *            The underlying reader that provides the original, not modified characters. For optimal performance
109     *            choose a {@link BufferedReader}.
110     * @param modifier
111     *            the object that modifies the stream.
112     */
113    public ModifyingReader(Reader reader, Modifier modifier) {
114        super();
115
116        this.delegate = reader;
117
118        // take injected properties
119        this.modifier = modifier;
120
121        // initialize the mutable state
122        this.numberOfCharactersToSkip = 0;
123        this.minimumLengthOfLookBehind = 0; // we cannot do anything else
124        this.firstModifiableCharacterInBuffer = 0;
125        this.requestedNumCharactersInBuffer = 1; // any value
126        this.characterBuffer = new StringBuilder(requestedNumCharactersInBuffer);
127        adjustCapacityOfBuffer();
128    }
129
130    /**
131     * Updates the input buffer according to {@link #minimumLengthOfLookBehind} and
132     * {@link #requestedNumCharactersInBuffer}, and then fills the buffer up to its capacity.
133     */
134    private void updateBuffer() throws IOException {
135
136        removeCharactersInBufferNotNeededAnyLonger();
137
138        adjustCapacityOfBuffer();
139
140        fill();
141    }
142
143    /**
144     * Deletes those characters at the start of the buffer we do not need any longer.
145     */
146    private void removeCharactersInBufferNotNeededAnyLonger() {
147
148        int charactersToDelete = firstModifiableCharacterInBuffer - minimumLengthOfLookBehind;
149
150        if (charactersToDelete > 0) {
151            characterBuffer.delete(0, charactersToDelete);
152            firstModifiableCharacterInBuffer -= charactersToDelete;
153        } else if (charactersToDelete < 0) {
154            onFaultyModifier(-52, charactersToDelete + " characters to delete but this is not possible ("
155                    + firstModifiableCharacterInBuffer + "," + minimumLengthOfLookBehind + ")");
156        }
157    }
158
159    /**
160     * Adjusts the capacity of the buffer.
161     */
162    protected void adjustCapacityOfBuffer() {
163
164        // is the current capacity not big enough?
165        if (characterBuffer.capacity() < requestedNumCharactersInBuffer) {
166
167            // increase the capacity (we delegate to the default behavior of
168            // StringBuffer)
169            characterBuffer.ensureCapacity(requestedNumCharactersInBuffer);
170
171        }
172
173    }
174
175    /**
176     * Reads more characters into the input buffer. This method will block until the buffer is filled with
177     * {@link #requestedNumCharactersInBuffer} characters , or an I/O error occurs, or the end of the stream is reached.
178     */
179    private void fill() throws IOException {
180
181        int length = requestedNumCharactersInBuffer - characterBuffer.length();
182        if (length <= 0) {
183            // nothing to do
184            return;
185        }
186
187        char[] buffer = new char[length];
188        int offset = 0;
189
190        while (length > 0) {
191
192            int readChars = delegate.read(buffer, offset, length);
193            if (readChars != -1) {
194                characterBuffer.append(buffer, offset, readChars);
195                offset += readChars;
196                length -= readChars;
197            } else {
198                endOfStreamHit = true;
199                break;
200            }
201
202        }
203
204    }
205
206    protected void onFaultyModifier(int errorCode, String errorMessage) {
207        // should we silently ignore any errors and fallback to a meaningful
208        // behavior? No because a faulty modifier should be fixed by a
209        // ValidModifier, i.e. the user must take the responsibility for the
210        // errors and deal with them. Therefore, we throw an exception here.
211
212        // (1) collect the current state of the involved objects
213        Map<String, Object> description = new HashMap<String, Object>();
214
215        // describe arguments
216        description.put("argument errorCode", errorCode);
217        description.put("argument errorMessage", errorMessage);
218
219        // describe reader
220        description.put("this", this);
221
222        // (2) create an error message
223        List<String> keys = new ArrayList<String>(description.keySet());
224        Collections.sort(keys);
225        StringBuilder sb = new StringBuilder();
226        for (String key : keys) {
227            sb.append("\n" + key + ": " + description.get(key));
228        }
229
230        // (3) throw exception
231        throw new FaultyModifierException(sb.toString(), description);
232    }

234    //
235    // interface Reader
236    //
237
238    /**
239     * TODO more doc
240     * 
241     * @see java.io.Reader#read(char[], int, int)
242     */
243    @Override
244    public int read(char[] cbuf, int off, int len) throws IOException {
245        int index = 0;
246        int read = 0;
247        while (index < len && (read = readCharacter()) != -1) {
248            cbuf[off + index] = (char) read;
249            index++;
250        }
251
252        if (index == 0 && read == -1) {
253            return -1;
254        } else {
255            return index;
256        }
257    }
258
259    /**
260     * Contract is similar to {@link java.io.Reader#read()}.
261     * <p>
262     * TODO more doc
263     */
264    protected int readCharacter() throws IOException {
265
266        if (numberOfCharactersToSkip == 0) {
267
268            boolean modifyAgainImmediately = false;
269
270            AfterModification afterModification = null;
271            do {
272
273                updateBuffer();
274
275                afterModification = modifier.modify(characterBuffer, firstModifiableCharacterInBuffer, endOfStreamHit);
276
277                lastAfterModificationForDebuggingOnly = afterModification;
278
279                // update minimumLengthOfLookBehind
280                minimumLengthOfLookBehind = afterModification.getNewMinimumLengthOfLookBehind();
281                if (minimumLengthOfLookBehind > firstModifiableCharacterInBuffer
282                        + afterModification.getNumberOfCharactersToSkip()) {
283                    onFaultyModifier(-11, "Requested Look Behind" + " is impossible because there are not enough "
284                            + "characters in the stream.");
285                }
286
287                // update requestedNumCharactersInBuffer
288                requestedNumCharactersInBuffer = minimumLengthOfLookBehind + afterModification.getNewNumberOfChars();
289                if (requestedNumCharactersInBuffer < minimumLengthOfLookBehind + 1) {
290                    onFaultyModifier(-13, "Requested Capacity" + " is two small because there must at least one"
291                            + " unread characters available after the " + "look behind characters characters in the"
292                            + " stream.");
293                }
294
295                modifyAgainImmediately = false;
296                // do we have to read at least a single character as there is no
297                // modifiable character left in the character buffer?
298                if (firstModifiableCharacterInBuffer >= characterBuffer.length() && !endOfStreamHit) {
299                    // yes, we need fresh input ->
300                    modifyAgainImmediately = true;
301                }
302                // has the modifier requested a modification at the current
303                // position?
304                else if (afterModification.isModifyAgainImmediately()) {
305                    modifyAgainImmediately = true;
306                }
307
308            } while (modifyAgainImmediately);
309
310            numberOfCharactersToSkip = afterModification.getNumberOfCharactersToSkip();
311
312            if (!afterModification.isModifyAgainImmediately() && numberOfCharactersToSkip == 0 && !endOfStreamHit) {
313                onFaultyModifier(-16, "Not a single characters shall be " + "skipped but this is not possible of "
314                        + "modifyAgain() returns false and the end of " + "stream is not reached yet.");
315            }
316            numberOfCharactersToSkip--;
317        } else {
318            numberOfCharactersToSkip--;
319        }
320
321        // is the end of the stream reached?
322        if (firstModifiableCharacterInBuffer >= characterBuffer.length()) {
323            // yes -> return -1
324            return -1;
325        } else {
326            // no -> return the next unread character
327            char result = characterBuffer.charAt(firstModifiableCharacterInBuffer);
328
329            firstModifiableCharacterInBuffer++;
330
331            return result;
332        }
333    }

339    /**
340     * @see java.lang.Object#toString()
341     */
342    @Override
343    public String toString() {
344        StringBuilder builder = new StringBuilder();
345        builder.append("ModifyingReader [\ncharacterBuffer=");
346        builder.append(characterBuffer);
347        builder.append(", \nnextUnreadCharacterInBuffer=");
348        builder.append(firstModifiableCharacterInBuffer);
349        builder.append(", \nmodifier=");
350        builder.append(modifier);
351        builder.append(", \nminimumLengthOfLookBehind=");
352        builder.append(minimumLengthOfLookBehind);
353        builder.append(", \nrequestedCapacityOfCharacterBuffer=");
354        builder.append(requestedNumCharactersInBuffer);
355        builder.append(", \nlockedCharacters=");
356        builder.append(numberOfCharactersToSkip);
357        builder.append(", \nendOfStreamHit=");
358        builder.append(endOfStreamHit);
359        builder.append(", \nlastModificationForDebuggingOnly=");
360        builder.append(lastAfterModificationForDebuggingOnly);
361        builder.append("]");
362        return builder.toString();
363    }
