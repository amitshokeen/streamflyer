017package com.github.rwitzel.streamflyer.xml;
018
019import java.util.regex.Matcher;
020import java.util.regex.Pattern;
021
022import org.apache.commons.io.input.XmlStreamReader;
023
024import com.github.rwitzel.streamflyer.core.AfterModification;
025import com.github.rwitzel.streamflyer.core.Modifier;
026import com.github.rwitzel.streamflyer.internal.thirdparty.ZzzValidate;
027import com.github.rwitzel.streamflyer.util.ModificationFactory;
028
029/**
030 * Replaces the XML version in the XML prolog with the given XML version. Adds an XML prolog if the stream does not
031 * contain an XML prolog.
032 * <p>
033 * <h1>Contents</h1>
034 * <p>
035 * <b> <a href="#g1">1. How and when do I use this modifier?</a><br/>
036 * <a href="#g2">2. Do I have to care about BOMs at the beginning of the stream?</a> <br/>
037 * <a href="g3">3. Is there any known limitation?</a> <br/>
038 * <a href="#g4">4. How much memory does the modifier consume?</a><br/>
039 * </b> <!-- ++++++++++++++++++++++++++++++ -->
040 * <p>
041 * <h3 id="g1">1. How and when do I use this modifier?</h3>
042 * <p>
043 * This modifier is an alternative to {@link InvalidXmlCharacterModifier} if you a have characters in an XML stream that
044 * are valid for XML 1.1 documents but invalid for XML 1.1 documents. In this case you use this modifier to change the
045 * XML version in the prolog of the document.
046 * <p>
047 * EXAMPLE:
048 * <code><pre class="prettyprint lang-java">// choose the input stream to modify
049ByteArrayInputStream inputStream = new ByteArrayInputStream(
050        bytesWithBom);
051
052// wrap the input stream by BOM skipping reader
053Reader reader = new XmlStreamReader(inputStream);
054
055// create the reader that changes the XML version to 1.1
056ModifyingReader modifyingReader = new ModifyingReader(reader,
057        new XmlVersionModifier("1.1", 8192));
058
059// use the modifying reader instead of the original reader
060String xml = IOUtils.toString(modifyingReader);
061
062assertTrue(xml.startsWith("&lt;?xml version='1.1'"));
063</pre></code>
064 * <h3 id="g2">2. Do I have to care about BOMs at the beginning of the stream?</h3>
065 * <p>
066 * Yes, you must use a BOM skipping reader that wraps the input stream. Apache's Commons IO {@link XmlStreamReader} does
067 * this for you.
068 * <h3 id="g3">3. Is there any known limitation?</h3>
069 * <p>
070 * Yes, this modifier throws a {@link XmlPrologRidiculouslyLongException} if the prolog of the XML document contains
071 * more than {@link #INITIAL_NUMBER_OF_CHARACTERS} characters. This can only happen if there is a lot of whitespace
072 * within the prolog, which is highly unlikely but not forbidden by the XML specification. You should know that even the
073 * <code>XmlReader</code> of Apache Commons which you probably use to detect the encoding cannot deal with such a kind
074 * of prolog.
075 * <h3 id="#g4">4. How much memory does the modifier consume?</h3>
076 * <p>
077 * The memory consumption of this modifier during the stream processing is roughly given by the second argument of
078 * {@link #XmlVersionModifier(String, int)} but the initial memory consumption is given by
079 * {@link #INITIAL_NUMBER_OF_CHARACTERS}.
080 * 
081 * @author rwoo
082 * @since 27.06.2011
083 */
084public class XmlVersionModifier implements Modifier {
085
086    //
087    // constants
088    //
089
090    public final int INITIAL_NUMBER_OF_CHARACTERS = 4096;
091
092    /**
093     * The internal state of {@link XmlVersionModifier}.
094     * <p>
095     * The state transitions are: from <code>INITIAL</code> to <code>PROLOG_REQUEST</code> to
096     * <code>NO_LONGER_MODIFYING</code>.
097     */
098    private enum XmlVersionModifierState {
099        /**
100         * The initial state. No input read yet.
101         */
102        INITIAL,
103
104        /**
105         * The modifier has requested to read the XML prolog.
106         */
107        PROLOG_REQUEST,
108
109        /**
110         * The modifier has read the XML prolog, modified it if necessary. Nothing more to do for the modifier.
111         */
112        NO_LONGER_MODIFYING
113    }
114
115    //
116    // injected properties
117    //
118
119    protected ModificationFactory factory;
120
121    protected String xmlVersion;
122
123    //
124    // properties that represent the mutable state
125    //
126
127    private XmlVersionModifierState state = XmlVersionModifierState.INITIAL;
128
129    //
130    // constructors
131    //
132
133    public XmlVersionModifier(String xmlVersion, int newNumberOfChars) {
134
135        ZzzValidate.notNull(xmlVersion, "xmlVersion must not be null");
136
137        this.factory = new ModificationFactory(0, newNumberOfChars);
138        this.xmlVersion = xmlVersion;
139    }
140
141    //
142    // Modifier.* methods
143    //
144
145    /**
146     * @see com.github.rwitzel.streamflyer.core.Modifier#modify(java.lang.StringBuilder, int, boolean)
147     */
148    @Override
149    public AfterModification modify(StringBuilder characterBuffer, int firstModifiableCharacterInBuffer,
150            boolean endOfStreamHit) {
151
152        switch (state) {
153
154        case NO_LONGER_MODIFYING:
155
156            return factory.skipEntireBuffer(characterBuffer, firstModifiableCharacterInBuffer, endOfStreamHit);
157
158        case INITIAL:
159
160            state = XmlVersionModifierState.PROLOG_REQUEST;
161
162            // you never know how many whitespace characters are in the prolog
163            return factory.modifyAgainImmediately(INITIAL_NUMBER_OF_CHARACTERS, firstModifiableCharacterInBuffer);
164
165        case PROLOG_REQUEST:
166
167            // (Should we do aware of BOMs here? No. I consider it the
168            // responsibility of the caller to provide characters without BOM.)
169
170            Matcher matcher = Pattern.compile("<\\?xml[^>]*version\\s*=\\s*['\"]((1.0)|(1.1))['\"].*").matcher(
171                    characterBuffer);
172            if (matcher.matches()) {
173
174                // replace version in prolog
175                characterBuffer.replace(matcher.start(1), matcher.end(1), xmlVersion);
176            } else {
177                // is there a prolog that is too long?
178                Matcher matcher2 = Pattern.compile("<\\?xml.*").matcher(characterBuffer);
179                if (matcher2.matches()) {
180                    // this is not normal at all -> throw exception
181                    throw new XmlPrologRidiculouslyLongException(characterBuffer.toString());
182                }
183
184                // insert prolog
185                characterBuffer.insert(0, "<?xml version='" + xmlVersion + "'>");
186            }
187
188            state = XmlVersionModifierState.NO_LONGER_MODIFYING;
189
190            return factory.skipEntireBuffer(characterBuffer, firstModifiableCharacterInBuffer, endOfStreamHit);
191
192        default:
193            throw new IllegalStateException("state " + state + " not supported");
194
195        }
196
197    }
198}
